2.5 Configuration File for Bare Metal Installation
===========================================
```
[TOPOLOGY]

MCenterAgent = localhost

[COMMON]

# Path to install PM MCenter inside
installation_path = /tmp

########################################## Database information ####################################

# Select DB. Currently we support mysql
# mcenter_db_type = mysql

# SQL DB hostname or ip
# sql_db_host = localhost

# SQL DB port
# sql_db_port = 3306

# SQL DB user/service account user name
# sql_db_user = root

# SQL DB password
# sql_db_password = mcenter

# SQL database name
# sql_db_database = mcenterdb

########################################### Zookeeper #############################################
# If you are using standalone Zookeeper running which is not installed by mcenter installer please
# provide ip/hostname:port for zookeeper_peers, while if you have zookeeper cluster please provide
# ip1:port1,ip2:port2...ipn:portn
###################################################################################################

# ZooKeeper host or hosts to use
# zookeeper_peers = localhost:2181
# or
# zookeeper_peers = 10.10.11.18:2181,10.10.11.19:2181

# zookeeper_data_dir = ${COMMON:installation_path}/mcenter/opensource-apps/zookeeper-<version>/data

######################################### LDAP Authentication #####################################

# Auth mode to use: internal or LDAP, default is internal
# auth_mode = internal
# auth_ldap_url = ldap://localhost:389
# auth_ldap_user_base_dn = "dc=example,dc=org"
# auth_ldap_group_base_dn = "cn=mcenter,dc=example,dc=org"

# For following key use either "uid" or "cn"
# auth_ldap_user_common_name_or_uid = "uid"
# auth_ldap_security_authentication = simple

########################################## MCenter Server Port ########################################

# MCenter server port
# mcenter_server_port = 3456

########################################## MCENTER Cert Details ########################################

# MCENTER server protocol, valid values are 'http'
# mcenter_server_protocol = http

########################################## JVM Options ############################################

# JVM options to pass to the JVM
# The example below will set the heap memory size to 8GB
# mcenter_server_jvm_options = -Xmx8192m

# JVM options to pass to the agent JVM
# The example below will set the heap memory size to 8GB
# mcenter_agent_jvm_options = -Xmx8192m

############################################# Instance ############################################

# Number of cores to use in the PM MCenter instance
# instance_cores = 4

# Number of megabytes to use by the instance
# instance_mem = 8192

# Number of megabytes to use by the instance job manager
# instance_jobmanager_mem = 8192

# Instance job manager web port
# instance_jobmanager_port = 10001

# Instance out file max size
# instance_out_file_size = 10M

# Instance out file max no. of files
# instance_num_of_out_files = 5

# Instance out file rotate interval in seconds - default 6 hours
# instance_out_file_rotate_interval = 21600
```
